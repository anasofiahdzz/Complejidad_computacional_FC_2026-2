\begin{enumerate}

    \item (\textit{2 puntos}) Sea $G=(V,E)$ una gráfica. Una $k$-\textit{coloración} de $G$ es una función $c:V\longrightarrow\{1,\ldots,k\}$ tal que:
    $$\forall u,v\in V\,\big[uv\in E\to c(u)\neq c(v)\big]$$
    Considera el lenguaje $\texttt{3COL}=\{Bin(G): G \text{ tiene una 3-coloración}\}$.
    \begin{enumerate}
        \item Dado que $G\in\texttt{3COL}$, ¿qué estructura propones como \textit{testigo} de este hecho?
        \item Muestra que el tamaño del testigo es polinomial en función de $|Bin(G)|$.
        \item Describe un algoritmo verificador $V$ que, dados $Bin(G)$ y el testigo, decida en tiempo polinomial si el testigo es válido.
        \item Concluye formalmente que $\texttt{3COL}\in\mathbf{NP}$.
        \item ¿Es posible que $\texttt{3COL}\in\mathbf{P}$? ¿Qué consecuencia tendría eso para las clases de complejidad?
    \end{enumerate}

    \item (\textit{1.5 puntos}) Sea $G$ una gráfica dirigida y ponderada, y $cost(P)$ la suma de los costos de las aristas de un circuito $P$. Considera:
    $$\texttt{TSP}=\{(G,B): \text{existe un circuito Hamiltoniano en $G$ con $cost(P)\leq B$}\}$$
    \begin{enumerate}
        \item ¿$\texttt{TSP}\in\mathbf{P}$? Justifica brevemente.
        \item Considera la familia:
        $$\texttt{TSP}^*=\{(G,P,B): \text{$P$ es un circuito Hamiltoniano en $G$ y $cost(P)\leq B$}\}$$
        Muestra que $\texttt{TSP}^*\in\mathbf{P}$ siguiendo los pasos:
        \begin{enumerate}[label=\roman*.]
            \item ¿Cómo verificas que $P$ visita cada vértice de $G$ exactamente una vez?
            \item ¿Cómo verificas que cada arista de $P$ existe en $G$?
            \item ¿Cómo calculas $cost(P)$ y verificas $cost(P)\leq B$?
            \item Concluye que el procedimiento anterior tiene complejidad polinomial.
        \end{enumerate}
    \end{enumerate}

    \item (\textit{2 puntos}) Dados $G=(V,E)$ y $S\subseteq V$, decimos que $S$ es una \textit{cubierta de vértices} de $G$ si:
    $$cover(G,S)\;\iff\;\forall uv\in E\,[u\in S\vee v\in S]$$
    Muestra que el siguiente lenguaje está en $\mathbf{NP}$:
    $$\texttt{VERTEXCOVER}=\Big\{(G,k):\exists S\subseteq V\,\big[|S|\leq k\wedge cover(G,S)\big]\Big\}$$
    \begin{enumerate}
        \item Propón un testigo para una instancia positiva $(G,k)\in\texttt{VERTEXCOVER}$.
        \item Muestra que el testigo tiene tamaño polinomial en $|(G,k)|$.
        \item Describe el verificador $V(G,k,S)$ y analiza su complejidad temporal.
        \item Concluye que $\texttt{VERTEXCOVER}\in\mathbf{NP}$.
    \end{enumerate}

    \item (\textit{2 puntos}) Dada una gráfica no dirigida $G=(V,E)$ y un entero $k$, define:
    $$\texttt{MAXCUT}=\Big\{(G,k):\exists S\subseteq V\,\Big[\,\big|\{uv\in E: |\{u,v\}\cap S|=1\}\big|\geq k\,\Big]\Big\}$$
    Muestra que $\texttt{MAXCUT}\in\mathbf{NP}$ siguiendo los pasos:
    \begin{enumerate}
        \item Propón un testigo para comprobar que $(G,k)\in\texttt{MAXCUT}$.
        \item Muestra que el testigo tiene tamaño polinomial en $|Bin(G,k)|$.
        \item Describe el testigo $u(G,k,S)$ y analiza la complejidad temporal de la verificación.
        \item Concluye que $\texttt{MAXCUT}\in\mathbf{NP}$.
    \end{enumerate}

    \item (\textit{1 punto}) Muestra que $\mathbf{P}\subseteq\mathbf{NP}\cap\mathbf{coNP}$.
    \begin{enumerate}
        \item Sea $L\in\mathbf{P}$. Construye un verificador polinomial para $L$ y concluye que $L\in\mathbf{NP}$.
        \item Muestra que $\overline{L}\in\mathbf{P}$ y usa el inciso anterior para concluir que $L\in\mathbf{coNP}$.
        \item Concluye que $\mathbf{P}\subseteq\mathbf{NP}\cap\mathbf{coNP}$.
    \end{enumerate}

    \item (\textit{1.5 puntos}) Muestra que si $\mathbf{P}=\mathbf{NP}$, entonces $\mathbf{NP}=\mathbf{coNP}$.
    \begin{enumerate}
        \item Asumiendo $\mathbf{P}=\mathbf{NP}$, sea $L\in\mathbf{NP}$. Muestra que $L\in\mathbf{coNP}$, y concluye $\mathbf{NP}\subseteq\mathbf{coNP}$.
        \item Bajo la misma hipótesis, sea $L\in\mathbf{coNP}$. Muestra que $L\in\mathbf{NP}$, y concluye $\mathbf{coNP}\subseteq\mathbf{NP}$.
        \item Concluye que $\mathbf{NP}=\mathbf{coNP}$.
    \end{enumerate}

\end{enumerate}
